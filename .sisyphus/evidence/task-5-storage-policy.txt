-- silver.dim_place
CREATE TABLE silver.dim_place\n(\n    `place_id` String,\n    `parent_place_id` Nullable(String),\n    `canonical_name` String,\n    `place_type` LowCardinality(String),\n    `admin_level` UInt8,\n    `country_code` LowCardinality(String),\n    `continent_code` LowCardinality(String),\n    `source_place_key` String,\n    `source_system` LowCardinality(String),\n    `status` LowCardinality(String),\n    `centroid_lat` Float64,\n    `centroid_lon` Float64,\n    `bbox_min_lat` Float64,\n    `bbox_min_lon` Float64,\n    `bbox_max_lat` Float64,\n    `bbox_max_lon` Float64,\n    `valid_from` DateTime64(3, \'UTC\'),\n    `valid_to` Nullable(DateTime64(3, \'UTC\')),\n    `schema_version` UInt32,\n    `record_version` UInt64,\n    `api_contract_version` UInt32,\n    `updated_at` DateTime64(3, \'UTC\'),\n    `attrs` String CODEC(ZSTD(3)),\n    `evidence` String CODEC(ZSTD(3))\n)\nENGINE = ReplacingMergeTree(record_version)\nORDER BY (country_code, place_type, place_id)\nSETTINGS index_granularity = 8192

-- silver.fact_observation
CREATE TABLE silver.fact_observation\n(\n    `observation_id` String,\n    `source_id` String,\n    `subject_type` LowCardinality(String),\n    `subject_id` String,\n    `observation_type` LowCardinality(String),\n    `place_id` String,\n    `parent_place_chain` Array(String),\n    `observed_at` DateTime64(3, \'UTC\'),\n    `published_at` Nullable(DateTime64(3, \'UTC\')),\n    `confidence_band` LowCardinality(String),\n    `measurement_unit` LowCardinality(String),\n    `measurement_value` Float64,\n    `schema_version` UInt32,\n    `attrs` String CODEC(ZSTD(3)),\n    `evidence` String CODEC(ZSTD(3))\n)\nENGINE = MergeTree\nPARTITION BY toYYYYMM(observed_at)\nORDER BY (place_id, observation_type, observed_at, observation_id)\nTTL toDateTime(observed_at) + toIntervalDay(1095)\nSETTINGS index_granularity = 8192

-- gold.metric_state
CREATE TABLE gold.metric_state\n(\n    `metric_id` String,\n    `subject_grain` LowCardinality(String),\n    `subject_id` String,\n    `place_id` String,\n    `window_grain` LowCardinality(String),\n    `window_start` DateTime64(3, \'UTC\'),\n    `window_end` DateTime64(3, \'UTC\'),\n    `contribution_count_state` AggregateFunction(count),\n    `contribution_value_state` AggregateFunction(sum, Float64),\n    `contribution_weight_state` AggregateFunction(sum, Float64),\n    `peak_value_state` AggregateFunction(max, Float64),\n    `last_contribution_at_state` AggregateFunction(max, DateTime64(3, \'UTC\')),\n    `updated_at` DateTime64(3, \'UTC\')\n)\nENGINE = AggregatingMergeTree\nPARTITION BY toYYYYMM(window_start)\nORDER BY (metric_id, subject_grain, subject_id, window_grain, window_start)\nTTL toDateTime(window_start) + toIntervalDay(730)\nSETTINGS index_granularity = 8192
